def balance_new(data,mean,max):
    data_copy = data.copy()
    x = 0

    ## So snh mean ca dataframe mi vi mean c
    if data_copy['Size'].mean() > mean and data_copy['Size'].mean() < max - 1000:
        new_mean = data_copy['Size'].mean()
    elif data_copy['Size'].mean() > mean and data_copy['Size'].mean() >= max - 1000:
        new_mean = mean
        display("Place more edge server for better performance")
    else:
        new_mean = mean

    # Chy vng lp  cn bng
    ### Tnh tng khong chnh lch vi mean
    for i in range(len(data_copy)):
        if data_copy['Size'].iloc[i] > mean:
            x += data_copy['Size'].iloc[i] - mean 
            data_copy['Size'].iloc[i] = mean

    # Cng nhng khong chnh lch  cho cc gi tr thiu.
    # Nhng m bo sum trc v sau khi cn bng s khng bng nhau.
    # V trong thc t. Cn b chnh lch
    for i in range(len(data_copy)):
        # Gn bin distance l khong chnh lch gia Size ti v tr i so vi mean
        distance = (mean-data_copy['Size'].iloc[i])

        # Trnh trng hp khong chnh lch x b m. Khc ny l chia u khong  cho cc s cn li
        if x - distance <= 0:
            while distance == 0:
                rest = data_copy['Size'][data_copy['Size'] < mean].to_frame()
                rest['Size'] += distance/len(rest)

        ## Ny l cng khong chnh lch cho ti khi x-distance <= 0
        elif x > 0 :
            if data_copy['Size'].iloc[i] < mean:
                data_copy['Size'].iloc[i] += (mean-data_copy['Size'].iloc[i])
                x = x - distance

    return data_copy,new_mean
